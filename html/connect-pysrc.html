107      logging.basicConfig( level=logging.INFO, format='%(levelname)s %(message)s' )

109   
110      HOST = "ahubswe.net" 
111      PORT = 2023 
112       
113      logging.info('Connectiong to %s:%s' % (HOST,PORT)) 
114      tn = telnetlib.Telnet(HOST,PORT) 
115      if tn == None: 
116          logging.error('Problem in connecting to the given host') 
117          sys.exit(-1) 
118      logging.debug('Connected') 
119   
120      # Read available data for one second 
121      s = tn.read_until( "xxxxxxxx" , 1) 
122      tn.close() 
123   
124      logging.debug('Received: %s' % s) 
125   
126      t = s.splitlines() 
127      for s in t: 
128          if not s or len(s) == 0: 
129              continue 
130          if s[0] == '#':    # Skip comment frames 
131              continue 
132          f = frame.Frame(s,"APRSD")
